startprogram
var integer n, i, num, sum, max, min;

n = 5;
sum = 0;
max = -10000;
min = 10000;

for i = 1 to n do
    read num;
    sum = sum + num;

    if num > max then
        max = num;
    endif

    if num < min then
        min = num;
    endif
endfor


write sum;
write max;
write min;
endprogram
